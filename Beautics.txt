Tables for system.

-- States Table
CREATE TABLE tblState (
    State_id INT PRIMARY KEY AUTO_INCREMENT,
    state_name VARCHAR(100),
    active BOOLEAN DEFAULT TRUE,
    is_disabled BOOLEAN DEFAULT FALSE
);

-- Cities Table
CREATE TABLE tblCity (
    City_id INT PRIMARY KEY AUTO_INCREMENT,
    city_name VARCHAR(100),
    state_id INT,
    active BOOLEAN DEFAULT TRUE,
    is_disabled BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (state_id) REFERENCES tblState(State_id)
);

-- Updated Users Table
CREATE TABLE tblUser (
    User_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255),
    contact VARCHAR(50),
    email VARCHAR(255),
    password VARCHAR(255),
    gender VARCHAR(10),
    flat_no VARCHAR(20),
    floor_no VARCHAR(10),
    building_name VARCHAR(100),
    road_street VARCHAR(150),
    pincode INT,
    city_id INT,
    profile_img VARCHAR(255),
    role VARCHAR(50),
    active BOOLEAN DEFAULT TRUE,
    is_disabled BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (city_id) REFERENCES tblCity(City_id)
);

-- Categories Table
CREATE TABLE tblCategory (
    Category_id INT PRIMARY KEY AUTO_INCREMENT,
    category_name VARCHAR(255),
    img_path VARCHAR(255),
    active BOOLEAN DEFAULT TRUE,
    is_disabled BOOLEAN DEFAULT FALSE
);

-- Products Table
CREATE TABLE tblProduct (
    Product_id INT PRIMARY KEY AUTO_INCREMENT,
    product_name VARCHAR(255),
    description TEXT,
    ingredients TEXT,
    category_id INT,
    active BOOLEAN DEFAULT TRUE,
    is_disabled BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (category_id) REFERENCES tblCategory(Category_id)
);

-- Colors Table
CREATE TABLE tblColor (
    Color_id INT PRIMARY KEY AUTO_INCREMENT,
    code VARCHAR(50),
    color_name VARCHAR(50),
    active BOOLEAN DEFAULT TRUE,
    is_disabled BOOLEAN DEFAULT FALSE
);

-- Variants Table
CREATE TABLE tblVariants (
    Variant_id INT PRIMARY KEY AUTO_INCREMENT,
    product_id INT,
    color_id INT,
    amount_type VARCHAR(50),
    amount_value DECIMAL(10, 2),
    price DECIMAL(10, 2),
    quantity INT,
    img_path VARCHAR(255),
    active BOOLEAN DEFAULT TRUE,
    is_disabled BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (product_id) REFERENCES tblProduct(Product_id),
    FOREIGN KEY (color_id) REFERENCES tblColor(Color_id)
);

-- Cart Table
CREATE TABLE tblCart (
    Cart_id INT PRIMARY KEY AUTO_INCREMENT,
    variant_id INT,
    user_id INT,    
    FOREIGN KEY (variant_id) REFERENCES tblVariants(Variant_id),
    FOREIGN KEY (user_id) REFERENCES tblUser(User_id)
);

-- Wishlist Table
CREATE TABLE tblWishlist (
    Wishlist_id INT PRIMARY KEY AUTO_INCREMENT,
    variant_id INT,
    user_id INT,
    FOREIGN KEY (variant_id) REFERENCES tblVariants(Variant_id),
    FOREIGN KEY (user_id) REFERENCES tblUser(User_id)
);

-- Orders Table
CREATE TABLE tblOrder (
    Order_id INT PRIMARY KEY AUTO_INCREMENT,
    variant_id INT,
    user_id INT,
    quantity INT,
    total_price DECIMAL(10, 2),
    order_date DATE,
    status VARCHAR(50), -- Paid,Confirmed (In progress), Dispatched, Delivered, Returned
    is_disabled BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (variant_id) REFERENCES tblVariants(Variant_id),
    FOREIGN KEY (user_id) REFERENCES tblUser(User_id)
);

-- Payments Table
CREATE TABLE tblPayment (
    Payment_id INT PRIMARY KEY AUTO_INCREMENT,
    order_id INT,
    payment_method VARCHAR(50),
    payment_amount DECIMAL(10, 2),
    payment_date DATETIME,
    is_disabled BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (order_id) REFERENCES tblOrder(Order_id)
);

-- Reviews Table
CREATE TABLE tblReview (
    Review_id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT,
    product_id INT,
    review_score INT CHECK (review_score BETWEEN 0 AND 5),
    created_at DATE,
    updated_at DATE,
    active BOOLEAN DEFAULT TRUE,
    is_disabled BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (user_id) REFERENCES tblUser(User_id),
    FOREIGN KEY (product_id) REFERENCES tblProduct(Product_id)
);
